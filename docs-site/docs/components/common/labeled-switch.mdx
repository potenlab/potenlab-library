# LabeledSwitch

A Switch paired with an inline clickable label.

## Demo

<StorybookEmbed storyId="common-labeled-switch--default" height={80} />

## Import

```tsx
import { LabeledSwitch } from "@potenlab/ui";
// or
import { LabeledSwitch } from "@potenlab/ui/components/common";
```

## Usage

```tsx
const [enabled, setEnabled] = useState(false);

<LabeledSwitch
  id="notifications"
  label="Enable notifications"
  checked={enabled}
  onCheckedChange={setEnabled}
/>;
```
